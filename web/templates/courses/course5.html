Раздел 1: Основы программирования микроконтроллеров

1.1. Введение в микроконтроллеры

Что такое микроконтроллер?

Микроконтроллер (МК) — это полноценная микрокомпьютерная система, интегрированная в единый кристалл. В отличие от микропроцессоров, которым требуются внешние компоненты, микроконтроллер содержит все необходимое для работы:

Ключевые компоненты МК:

  • Центральный процессор (CPU) — выполняет программный код
  • Память программ (Flash) — хранит прошивку (обычно 8KB-2MB)
  • Оперативная память (SRAM) — временное хранение данных (1KB-256KB)
  • Периферийные устройства:
    • GPIO (цифровые входы/выходы)
    • АЦП/ЦАП (аналогово-цифровые преобразователи)
    • Таймеры/счетчики
    • Интерфейсы связи (UART, SPI, I2C, USB)

Популярные семейства микроконтроллеров:

1. AVR (Arduino)
  • ATmega328P (Arduino Uno/Nano)
  • Особенности:
    • 8-битная архитектура
    • 16 МГц тактовая частота
    • 32 KB Flash, 2 KB SRAM
    • Простота освоения
2. ARM Cortex (STM32)
  • STM32F103 (Blue Pill)
  • Особенности:
    • 32-битная архитектура
    • До 72 МГц
    • 64-256 KB Flash
    • Богатая периферия
3. ESP32
  • Двухъядерный процессор
  • Встроенные Wi-Fi и Bluetooth
  • 160-240 МГц
  • Идеален для IoT проектов

Детальный разбор кода:

        
        // Функция setup выполняется один раз при старте
        void setup() {
        // Инициализация цифрового вывода 13 (встроенный светодиод)
        pinMode(LED_BUILTIN, OUTPUT);
        }

        // Функция loop выполняется циклически
        void loop() {
        digitalWrite(LED_BUILTIN, HIGH); // Подаем высокий уровень (5V)

        digitalWrite(LED_BUILTIN, LOW); // Подаем низкий уровень (0V)
        delay(1000); // Снова пауза

Практическое задание

Цель: Создать программу, которая:

  1. Мигает встроенным светодиодом с разной частотой
  2. Выводит счетчик в Serial Monitor
  3. Реагирует на внешнюю кнопку

Пример решения:


        int counter = 0;
        bool fastMode = false;

        void setup() {
        pinMode(LED_BUILTIN, OUTPUT);
        pinMode(2, INPUT_PULLUP); // Кнопка на пине 2
        Serial.begin(9600);
        }

        void loop() {
        if(digitalRead(2) == LOW) {
        fastMode = !fastMode;
        delay(200); // Защита от дребезга
        }

        digitalWrite(LED_BUILTIN, HIGH);
        Serial.print("Counter: ");
        Serial.println(counter++);

        if(fastMode) {
        delay(100); // Быстрый режим
        } else {
        delay(500); // Медленный режим
        }

        digitalWrite(LED_BUILTIN, LOW);
        delay(fastMode ? 100 : 500);
    }

2.1. Основы синтаксиса и структуры программ

Структура программы Arduino:

Каждая программа (скетч) для Arduino состоит из двух обязательных функций:


        void setup() {
            // Инициализация параметров - выполняется один раз при старте
            pinMode(13, OUTPUT); // Настройка пина 13 как выхода
            Serial.begin(9600); // Инициализация последовательного порта
            }

            void loop() {
            // Основной цикл - выполняется бесконечно
            digitalWrite(13, HIGH);
            delay(500);
            digitalWrite(13, LOW);
            delay(500);
        }

Раздел 3: Практическое программирование микроконтроллеров

3.1 Работа с цифровыми входами/выходами

Настройка GPIO (General Purpose Input/Output):


        void setup() {
            pinMode(13, OUTPUT); // Настройка пина 13 как выхода
            pinMode(2, INPUT_PULLUP); // Настройка пина 2 как входа с подтяжкой
        }

Основные функции:

  • digitalWrite(pin, HIGH/LOW) — установка состояния выхода
  • digitalRead(pin) — чтение состояния входа
  • pinMode(pin, mode) — настройка режима работы пина

Практический пример: Управление светодиодом кнопкой


        const int buttonPin = 2;
        const int ledPin = 13;
        int buttonState = 0;

        void setup() {
            pinMode(ledPin, OUTPUT);
            pinMode(buttonPin, INPUT_PULLUP);
        }

        void loop() {
            buttonState = digitalRead(buttonPin);
            if (buttonState == LOW) {
                digitalWrite(ledPin, HIGH);
            } else {
                digitalWrite(ledPin, LOW);
            }
        }

3.2 Аналоговые входы и ШИМ

Аналого-цифровое преобразование (АЦП):


        int sensorValue = analogRead(A0); // Чтение значения (0-1023)
        float voltage = sensorValue * (5.0 / 1023.0); // Преобразование в вольты

ШИМ (PWM) выходы:

analogWrite(9, 128); // 50% заполнение (0-255)

Пример: Плавное изменение яркости светодиода


        int brightness = 0;
        int fadeAmount = 5;

        void setup() {
        pinMode(9, OUTPUT);
        }

        void loop() {
        analogWrite(9, brightness);
        brightness += fadeAmount;
        if (brightness <= 0 || brightness >= 255) {
        fadeAmount = -fadeAmount;
        }
        delay(30);
    }

3.3 Работа с таймерами и прерываниями

Настройка прерываний:


        attachInterrupt(digitalPinToInterrupt(2), buttonPressed, FALLING);
        void buttonPressed() {
            // Обработчик прерывания
        }

Типы прерываний:

  • RISING — при переходе из LOW в HIGH
  • FALLING — при переходе из HIGH в LOW
  • CHANGE — при любом изменении состояния

Пример использования таймера:


        unsigned long previousMillis = 0;
        const long interval = 1000; // Интервал 1 секунда

        void loop() {
            unsigned long currentMillis = millis();
            if (currentMillis - previousMillis >= interval) {
                previousMillis = currentMillis;
            digitalWrite(13, !digitalRead(13)); // Переключение светодиода
        }
    }

Проект 2: Система автоматического полива


        #include "DHT.h"
        #define DHTPIN 2
        #define DHTTYPE DHT11
        #define PUMP_PIN 3
        #define SOIL_MOISTURE_PIN A0

        DHT dht(DHTPIN, DHTTYPE);

        void setup() {
            pinMode(PUMP_PIN, OUTPUT);
            Serial.begin(9600);
            dht.begin();
        }

        void loop() {
            float humidity = dht.readHumidity();
            int soilMoisture = analogRead(SOIL_MOISTURE_PIN);
            if (soilMoisture < 300 && humidity < 70) {
                digitalWrite(PUMP_PIN, HIGH);
                delay(5000);
                digitalWrite(PUMP_PIN, LOW);
            }
            delay(3600000); // Проверка каждый час
        }
